# this is a script to run METAL meta-analysis software. Two gwases against glioma subtypes  were conducted on AGS_illumina and AGS_Onco cohorts and will be used in this analysis.
# define meta-analysis scheme to use effect-sizes (beta) and standard error
SCHEME STDERR
# define separation format of input files
SEPARATOR COMMA
# enable genomic correction (adjusts test stats to account for small amounts of population stratification or unaccounted for relatedness)
GENOMICCONTROL OFF
# enable tracking of allele frequencies across all files
#AVERAGEFREQ ON
#MINMAXFREQ ON

# describe and process first input file
MARKER pgs
#ALLELE ALT REF
EFFECT coef
STDERR se_coef
PVALUE pvalue
#FREQLABEL AF
PROCESS /francislab/data1/working/20250800-AGS-CIDR-ONCO-I370-TCGA/20250724-pgs/pgs-onco-hg19/onco_SUBSETNAME/onco_SUBSETNAME.cox_coeffs.csv

# describe and process second input file
MARKER pgs
#ALLELE ALT REF
EFFECT coef
STDERR se_coef
PVALUE pvalue
#FREQLABEL AF
PROCESS /francislab/data1/working/20250800-AGS-CIDR-ONCO-I370-TCGA/20250724-pgs/pgs-i370-hg19/i370_SUBSETNAME/i370_SUBSETNAME.cox_coeffs.csv

# describe and process second input file
MARKER pgs
#ALLELE ALT REF
EFFECT coef
STDERR se_coef
PVALUE pvalue
#FREQLABEL AF
PROCESS /francislab/data1/working/20250800-AGS-CIDR-ONCO-I370-TCGA/20250724-pgs/pgs-tcga-hg19/tcga_SUBSETNAME/tcga_SUBSETNAME.cox_coeffs.csv

#	later add the cidr dataset

# run analysis
OUTFILE /francislab/data1/working/20250800-AGS-CIDR-ONCO-I370-TCGA/20250724-pgs/metal_survival_SUBSETNAME_ .tbl
ANALYZE HETEROGENEITY
QUIT
